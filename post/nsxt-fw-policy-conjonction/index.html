resource "nsxt_policy_group" "grp" {
    for_each = var.map_grp
    display_name = each.key

    dynamic "criteria" {
        #The for_each contain a for loop with filter to create the EXPRESSIONS criteria
        for_each = merge(element(each.value, var.EXP)...)
        content {
            member_type = element(criteria.value, var.EXP_MTYPE)
            key = element(criteria.value, var.EXP_KEY)
            operator = element(criteria.value, var.EXP_OP)
            value = condition.value
        }
    }

    dynamic "criteria" {
        #The for_each contain a for loop with filter to create the IP criteria
        for_each = merge(element(each.value, var.IP)...)
        content {
            ipaddress_expression {
                ip_addresses = criteria.value
            }
        }
    }

    dynamic "criteria" {
        #The for_each contain a for loop with filter to create the MAC criteria
        for_each = merge(element(each.value, var.MAC)...)
        content {
            ipaddress_expression {
                mac_addresses = criteria.value
            }
        }
    }

    dynamic "conjunction" {
        #The for_each contain a for loop with filter to create the criteria conjunction
        for_each = merge(element(each.value, var.CNJ)...)
        content {
            operator = conjunction.value
        }
    }
}


variable "map_grp" {
    type = map(any)
    default = {
    }
}
